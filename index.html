netflu_example.nf
func main() {
  let greeting = "Hello, Netflu!"
  let numbers = [1, 2, 3, 4, 5]
  
  for num in numbers {
    print(greeting + " 数字: " + num)
  }
  
  let sum = numbers.reduce(0) { acc, num => acc + num }
  print("总和: " + sum)
  
  return 0
}

安装 Netflu

# macOS/Linux 通过包管理器安装
curl -sSf https://netflu-lang.org/install.sh | sh

# 或者手动下载二进制文件
wget https://netflu-lang.org/releases/netflu-latest-x86_64-unknown-linux-gnu.tar.gz
tar -xzf netflu-latest-x86_64-unknown-linux-gnu.tar.gz
sudo mv netflu /usr/local/bin/

# Windows 通过 PowerShell 安装
iwr https://netflu-lang.org/install.ps1 -useb | iex

创建第一个程序

# 创建新项目
netflu new hello-world
cd hello-world

# 编辑 main.nf 文件
func main() {
  print("Hello, Netflu!")
  return 0
}

# 编译并运行
netflu build
./hello-world

web_server.nf
import http

func main() {
  let server = http.Server(port: 8080)
  
  server.get("/") { req, res =>
    res.send("Hello, Netflu Web Server!")
  }
  
  server.get("/json") { req, res =>
    res.json({
      message: "这是一个JSON响应",
      data: [1, 2, 3, 4, 5]
    })
  }
  
  print("服务器运行在 http://localhost:8080")
  server.start()
  return 0
}

data_processing.nf
import csv
import math

func main() {
  let data = csv.read("data.csv")
  
  let ages = data.map { row => row.get("age").parseInt() }
  let avg_age = ages.reduce(0) { acc, age => acc + age } / ages.len()
  
  let filtered = data.filter { row =>
    row.get("age").parseInt() > 30
  }
  
  csv.write("filtered_data.csv", filtered)
  
  print("平均年龄: " + avg_age.toString())
  print("过滤后的数据已保存到 filtered_data.csv")
  
  return 0
}

async_example.nf
import http
import time

async func fetchData(url: string) {
  let response = await http.getAsync(url)
  return response.body
}

async func main() {
  let start = time.now()
  
  let [data1, data2, data3] = await Promise.all([
    fetchData("https://api.example.com/data1"),
    fetchData("https://api.example.com/data2"),
    fetchData("https://api.example.com/data3")
  ])
  
  print(data1.length)
  print(data2.length)
  print(data3.length)
  
  let end = time.now()
  print("总耗时: " + (end - start).toString() + "ms")
  
  return 0
}

oop_example.nf
class Shape {
  public color: string
  
  constructor(color: string) {
    this.color = color
  }
  
  public func describe() {
    print("这是一个" + this.color + "的形状")
  }
  
  public abstract func area(): number
}

class Circle extends Shape {
  private radius: number
  
  constructor(color: string, radius: number) {
    super(color)
    this.radius = radius
  }
  
  public override func area() {
    return math.PI * this.radius * this.radius
  }
}

func main() {
  let circle = new Circle("红色", 5)
  circle.describe()
  print("面积: " + circle.area().toString())
  
  return 0
}
